void setupClassificationModel() {
  static tflite::MicroErrorReporter error_reporter;
  static tflite::MicroMutableOpResolver<6> resolver;
  resolver.AddFullyConnected();
  resolver.AddRelu();
  resolver.AddLogistic();
  
  const tflite::Model* model = tflite::GetModel(classification_model_tflite);
  if (model->version() != TFLITE_SCHEMA_VERSION) {
      Serial.println("Model schema version mismatch!");
      return;
  }

  classificationInterpreter = new tflite::MicroInterpreter(
      model, resolver, tensor_arena, kTensorArenaSize, &error_reporter);
  
      classificationInterpreter->AllocateTensors();
}

// ~~~~~~~~~~~~~~~~~~~~~Connect to Losant Function~~~~~~~~~~~~~~~~~~~~~~~~~~
void connectToLosant() {
  Serial.print("Connecting to WiFi...");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println(" Connected!");

  secureClient.setInsecure();  // Use for testing. Replace with certificate verification in production.
  mqttClient.setServer("broker.losant.com", 8883);

  while (!mqttClient.connected()) {
    Serial.print("Connecting to Losant...");
    if (mqttClient.connect(LOSANT_DEVICE_ID, LOSANT_ACCESS_KEY, LOSANT_ACCESS_SECRET)) {
      Serial.println(" Connected to Losant!");
    } else {
      Serial.print(" Failed, rc=");
      Serial.print(mqttClient.state());
      Serial.println(" Retrying...");
      delay(2000);
    }
  }
} //~~~~~~~~~~~~~~~~~~~~~~~~~~~~end of function~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

void setup() {

  My_LCD.init();
  My_LCD.backlight();
  My_LCD.clear();
  My_LCD.print("Initializing");
  My_LCD.setCursor(0, 1);
  My_LCD.print("GSW-01...");

  Serial.begin(9600);    // Set serial baud rate to 9600
  delay(500);
  while (!Serial) { ; }  // wait for serial port to connect. Needed for native USB port only
  
  connectToLosant(); //~~~~~~~~~~~~Calls connectToLosant function

  Serial.println("Initializing SD card...");
  if (!SD.begin(CS)) {
    Serial.println("initialization failed!");
    return;
  }
  Serial.println("initialization done.");
  
  myFile = SD.open("/GSWdata.txt", FILE_WRITE);
  myFile.println("CO, Alcohol, CO2, Toluene, NH4, Acetone");
  myFile.close();

  //Set math model to calculate the PPM concentration and the value of constants
  MQ135.setRegressionMethod(1); //_PPM =  a*ratio^b
  MQ135.setA(102.2); MQ135.setB(-2.473); // Configure the equation to to calculate NH4 concentration
  MQ135.init(); 
  float calcR0 = 0;
  for(int i = 1; i<=10; i ++)
  {
    MQ135.update(); // Update data, the arduino will read the voltage from the analog pin
    calcR0 += MQ135.calibrate(RatioMQ135CleanAir);
    // Serial.print(".");
  }
  MQ135.setR0(calcR0/10);

  pinMode(buzzerPin, OUTPUT);
  pinMode(redPin, OUTPUT);
  pinMode(bluePin, OUTPUT);

  //Menu buttons initialization
  pinMode(upButton, INPUT_PULLUP);
  pinMode(downButton, INPUT_PULLUP);
  pinMode(selButton, INPUT_PULLUP);

  digitalWrite(bluePin, HIGH);
  digitalWrite(redPin, HIGH);
  tone(buzzerPin, 180, 500);
  delay(500);
  noTone(buzzerPin); // Stop tone after initial beep
  digitalWrite(redPin, LOW);

}


void loop() {


  MQ135.update(); // Update data, the arduino will read the voltage from the analog pin

  MQ135.setA(605.18); MQ135.setB(-3.937); // Configure the equation to calculate CO concentration value
  float CO = MQ135.readSensor(); // Sensor will read PPM concentration using the model, a and b values set previously or from the setup

  MQ135.setA(77.255); MQ135.setB(-3.18); //Configure the equation to calculate Alcohol concentration value
  float Alcohol = MQ135.readSensor(); // SSensor will read PPM concentration using the model, a and b values set previously or from the setup

  MQ135.setA(110.47); MQ135.setB(-2.862); // Configure the equation to calculate CO2 concentration value
  float CO2 = MQ135.readSensor(); // Sensor will read PPM concentration using the model, a and b values set previously or from the setup
  CO2 = CO2 + 400; //Assumes CO2 atmosphere is at least 400ppm

  MQ135.setA(44.947); MQ135.setB(-3.445); // Configure the equation to calculate Toluen concentration value
  float Toluen = MQ135.readSensor(); // Sensor will read PPM concentration using the model, a and b values set previously or from the setup
  
  MQ135.setA(102.2 ); MQ135.setB(-2.473); // Configure the equation to calculate NH4 concentration value
  float NH4 = MQ135.readSensor(); // Sensor will read PPM concentration using the model, a and b values set previously or from the setup

      menuState--;

    //default screen
    //if (menuState != menuLast) {

          My_LCD.setCursor(0, 0);

          My_LCD.setCursor(0, 1);
          My_LCD.print("Alcohol: ");
          My_LCD.print(Alcohol);
          My_LCD.print("ppm");
          break;
        case 1:
          My_LCD.setCursor(0, 0);
          My_LCD.print("Alcohol: ");
          My_LCD.print(Alcohol);
          My_LCD.print("ppm");
          My_LCD.setCursor(0, 1);
          My_LCD.print("CO2: ");
          My_LCD.print(CO2);
          My_LCD.print("ppm");
          break;
        case 2:
          My_LCD.setCursor(0, 0);
          My_LCD.print("CO2: ");
          My_LCD.print(CO2);
          My_LCD.print("ppm");
          My_LCD.setCursor(0, 1);
          My_LCD.print("Toluene: ");
          My_LCD.print(Toluen);
          My_LCD.print("ppm");
          break;
        case 3:
          My_LCD.setCursor(0, 0);
          My_LCD.print("Toluene: ");
          My_LCD.print(Toluen);
          My_LCD.print("ppm");
          My_LCD.setCursor(0, 1);
          My_LCD.print("NH4: ");
          My_LCD.print(NH4);
          My_LCD.print("ppm");
          break;
        case 4:
          My_LCD.setCursor(0, 0);
          My_LCD.print("NH4: ");
          My_LCD.print(NH4);
          My_LCD.print("ppm");
          My_LCD.setCursor(0, 1);

  if (millis() - lastSendTime > sendInterval) {

    payload += "\"CO2\":" + String(CO2) + ",";  // Offset applied
    payload += "\"Toluene\":" + String(Toluen) + ",";
    payload += "\"NH4\":" + String(NH4) + ",";
    payload += "\"Acetone\":" + String(Aceton);
    payload += "}";

    String topic = "losant/" + String(LOSANT_DEVICE_ID) + "/state";

    // Wrap payload in Losant format
    String losantPayload = "{\"data\":" + payload + "}";

    // Send to Losant

    if (!mqttClient.connected()) {
      connectToLosant();
    }


    mqttClient.publish(topic.c_str(), losantPayload.c_str());
    Serial.println("Published to Losant: " + losantPayload);
  }
/*****************************************************/

/*------------Trigger Alarm-------------------*/
  if (CO > 500 || NH4 > 300 || Alcohol > 400 || CO2 > 40000 || Toluen > 500 || Aceton > 1000) {

      noTone(buzzerPin);
    }

      digitalWrite(redPin, HIGH);

    else {
